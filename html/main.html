碰撞时候的速度变化

function [v11,v22] = Crash(v1,v2,m1,m2)
    v11 = ((m1-m2)*v1+2*m2*v2)/(m1+m2);
    v22 = ((m2-m1)*v2+2*m1*v1)/(m1+m2);
end
